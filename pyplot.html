import matplotlib.pyplot as plt
import random

# Generate a sequence of at least 5 random numbers
random_numbers = [random.randint(1, 100) for _ in range(5)]

# Create a figure and two subplots
fig, (ax1, ax2) = plt.subplots(2, 1, figsize=(8, 6))

# Visualize the data using scatter plot
ax1.scatter(range(len(random_numbers)), random_numbers, color='red')
ax1.set_title('Scatter Plot')
ax1.set_xlabel('Index')
ax1.set_ylabel('Random Numbers')

# Visualize the data using line plot (plot)
ax2.plot(range(len(random_numbers)), random_numbers, color='blue')
ax2.set_title('Line Plot')
ax2.set_xlabel('Index')
ax2.set_ylabel('Random Numbers')

# Adjust layout to prevent overlap
plt.tight_layout()

# Show the plots
plt.show()
